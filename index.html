async function generate() { const fileInput = document.querySelector('input[type="file"]'); const file = fileInput.files[0]; // Seçili stil butonunu bul (Örn: Chibi, Pixar vb.) const style = document.querySelector('.selected-style')?.innerText || "Pixar"; if(!file) return alert("Lütfen bir fotoğraf yükleyin!"); const loading = document.getElementById('loading'); const imgElement = document.getElementById('resultImage'); // Sağdaki sonuç alanı if(loading) loading.style.display = 'block'; const reader = new FileReader(); reader.readAsDataURL(file); reader.onload = async () => { const base64Image = reader.result.split(',')[1]; try { const response = await fetch('/api/generate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ image: base64Image, style: style }) }); const data = await response.json(); if (data.imageUrl) { imgElement.src = data.imageUrl; imgElement.style.display = 'block'; } else { alert("Hata: " + (data.error || "Bakiye yetersiz veya servis meşgul.")); } } catch (e) { alert("Bağlantı hatası!"); } finally { if(loading) loading.style.display = 'none'; } }; }
